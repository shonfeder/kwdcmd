Module Kwdcmd.Exec

The default CLI executors, derived use the Default_handlers for handling exits and errors.

If your program can produce errors other than those of type `Msg of string, then you should override Exec_handler.err_handler.

val commands : ?help:Stdlib.Format.formatter -> ?err:Stdlib.Format.formatter -> ?catch:bool -> ?env:(string -> string option) -> ?argv:string array -> ?default:(('a, [> `Msg of string ] as 'b) Stdlib.result Cmdliner.Term.t * Cmdliner.Term.info) -> ?doc:string -> ?sdocs:string -> ?exits:Cmdliner.Term.exit_info list -> ?man:Cmdliner.Manpage.block list -> ?version:string -> name:string -> (('a, 'b) Stdlib.result Cmdliner.Term.t * Cmdliner.Term.info) list -> unit

Subcommand selector entrypoint.

Example usage:

open Kwdcmd

let mode = Optional.(flag_choice ~default:"mode-a"
                      [ c ~name:"a" "mode-a" ~doc:"use mode a"
                      ; c ~name:"b" "mode-b" ~doc:"use mode b"
                      ])


let () = commands ~name:"myprog" ~version:"9.0.0"
    [ ( cmd ~name:"add" ~doc:"add stuff"
        @@ let+ stuff =
             Required.pos
               "STUFF"
               ~conv:Arg.string
               ~nth:0
               ~doc:"The stuff to add"
        and+ mode = mode
        in
        adder stuff mode )
    ; ( cmd ~name:"remove" ~doc:"remove things"
        @@ let+ stuff =
             Required.pos
               "THINGS"
               ~conv:Arg.string
               ~nth:1
               ~doc:"The stuff to remove"
        and+ mode = mode
        in
        remover stuff mode )
    ]
